## PART 4 — Client handler

```js
async function runSelectedScheduler() {
  const id = selectedJobId;

  const status = document.getElementById("status-box");
  status.textContent = "Running via scheduler…";

  try {
    const res = await fetch(`/run-selected-scheduler?id=${id}`, {
      method: "POST"
    });

    const data = await res.json();
    status.textContent = data.message;
  } catch (e) {
    status.textContent = "Error running scheduler flow";
  }
}
```
