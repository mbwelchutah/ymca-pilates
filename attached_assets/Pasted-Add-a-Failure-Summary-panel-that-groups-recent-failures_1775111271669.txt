# PART 1 — Extend /api/failures response

Modify existing endpoint to ALSO return counts:

```js
app.get("/api/failures", (req, res) => {
  const dir = "screenshots";

  if (!fs.existsSync(dir)) {
    return res.json({ recent: [], summary: {} });
  }

  const files = fs.readdirSync(dir)
    .filter(f => f.endsWith(".png"))
    .map(name => {
      const parts = name.split("-");
      const reason = parts[2]; // job-2-time-...
      const time = fs.statSync(path.join(dir, name)).mtime;

      return { name, reason, time };
    });

  // sort newest first
  const sorted = files.sort((a, b) => b.time - a.time);

  // last 5 for UI
  const recent = sorted.slice(0, 5);

  // group counts
  const summary = {};

  for (const f of files) {
    summary[f.reason] = (summary[f.reason] || 0) + 1;
  }

  res.json({ recent, summary });
});
```

## PART 5 — Summary renderer

```js
function renderSummary(summary) {
  const el = document.getElementById("failure-summary");
  el.innerHTML = "";

  const labels = {
    "time": "Time mismatch",
    "instructor": "Instructor mismatch",
    "time-instructor": "Time + Instructor"
  };

  for (const key in summary) {
    const item = document.createElement("div");
    item.className = "summary-item";

    item.innerHTML = `
      <span>${labels[key] || key}</span>
      <span class="summary-count">${summary[key]}</span>
    `;

    el.appendChild(item);
  }
}
```
